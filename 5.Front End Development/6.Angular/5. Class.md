## Class

    class Employee{
       name! : string;
       id! : number;
       address! : string;
    }
    let john = new Employee();
    john.id = 1;
    john.name = "Doe";
    john.address = "kolkata";
    console.log(john);  // Employee { id: 1, name: 'Doe', address: 'kolkata' }
